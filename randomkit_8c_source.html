140 void
141 rk_seed(unsigned long seed, rk_state *state)
142 {
143  int pos;
144  seed &= 0xffffffffUL;
145 
146  /* Knuth's PRNG as used in the Mersenne Twister reference implementation */
147  for (pos = 0; pos < RK_STATE_LEN; pos++) {
148  state->key[pos] = seed;
149  seed = (1812433253UL * (seed ^ (seed >> 30)) + pos + 1) & 0xffffffffUL;
150  }
151  state->pos = RK_STATE_LEN;
152  state->gauss = 0;
153  state->has_gauss = 0;
154  state->has_binomial = 0;
155 }

157 /* Thomas Wang 32 bits integer hash function */
158 unsigned long
159 rk_hash(unsigned long key)
160 {
161  key += ~(key << 15);
162  key ^= (key >> 10);
163  key += (key << 3);
164  key ^= (key >> 6);
165  key += ~(key << 11);
166  key ^= (key >> 16);
167  return key;
168 }

213 /* Slightly optimised reference implementation of the Mersenne Twister */
214 unsigned long
215 rk_random(rk_state *state)
216 {
217  unsigned long y;
218 
219  if (state->pos == RK_STATE_LEN) {
220  int i;
221 
222  for (i = 0; i < N - M; i++) {
223  y = (state->key[i] & UPPER_MASK) | (state->key[i+1] & LOWER_MASK);
224  state->key[i] = state->key[i+M] ^ (y>>1) ^ (-(y & 1) & MATRIX_A);
225  }
226  for (; i < N - 1; i++) {
227  y = (state->key[i] & UPPER_MASK) | (state->key[i+1] & LOWER_MASK);
228  state->key[i] = state->key[i+(M-N)] ^ (y>>1) ^ (-(y & 1) & MATRIX_A);
229  }
230  y = (state->key[N - 1] & UPPER_MASK) | (state->key[0] & LOWER_MASK);
231  state->key[N - 1] = state->key[M - 1] ^ (y >> 1) ^ (-(y & 1) & MATRIX_A);
232 
233  state->pos = 0;
234  }
235  y = state->key[state->pos++];
236 
237  /* Tempering */
238  y ^= (y >> 11);
239  y ^= (y << 7) & 0x9d2c5680UL;
240  y ^= (y << 15) & 0xefc60000UL;
241  y ^= (y >> 18);
242 
243  return y;
244 }

262 unsigned long
263 rk_interval(unsigned long max, rk_state *state)
264 {
265  unsigned long mask = max, value;
266 
267  if (max == 0) {
268  return 0;
269  }
270  /* Smallest bit mask >= max */
271  mask |= mask >> 1;
272  mask |= mask >> 2;
273  mask |= mask >> 4;
274  mask |= mask >> 8;
275  mask |= mask >> 16;
276 #if ULONG_MAX > 0xffffffffUL
277  mask |= mask >> 32;
278 #endif
279 
280  /* Search a random value in [0..mask] <= max */
281 #if ULONG_MAX > 0xffffffffUL
282  if (max <= 0xffffffffUL) {
283  while ((value = (rk_random(state) & mask)) > max);
284  }
285  else {
286  while ((value = (rk_ulong(state) & mask)) > max);
287  }
288 #else
289  while ((value = (rk_ulong(state) & mask)) > max);
290 #endif
291  return value;
292 }

294 double
295 rk_double(rk_state *state)
296 {
297  /* shifts : 67108864 = 0x4000000, 9007199254740992 = 0x20000000000000 */
298  long a = rk_random(state) >> 5, b = rk_random(state) >> 6;
299  return (a * 67108864.0 + b) / 9007199254740992.0;
300 }

302 void
303 rk_fill(void *buffer, size_t size, rk_state *state)
304 {
305  unsigned long r;
306  unsigned char *buf = buffer;
307 
308  for (; size >= 4; size -= 4) {
309  r = rk_random(state);
310  *(buf++) = r & 0xFF;
311  *(buf++) = (r >> 8) & 0xFF;
312  *(buf++) = (r >> 16) & 0xFF;
313  *(buf++) = (r >> 24) & 0xFF;
314  }
315 
316  if (!size) {
317  return;
318  }
319  r = rk_random(state);
320  for (; size; r >>= 8, size --) {
321  *(buf++) = (unsigned char)(r & 0xFF);
322  }
323 }

379 double
380 rk_gauss(rk_state *state)
381 {
382  if (state->has_gauss) {
383  const double tmp = state->gauss;
384  state->gauss = 0;
385  state->has_gauss = 0;
386  return tmp;
387  }
388  else {
389  double f, x1, x2, r2;
390 
391  do {
392  x1 = 2.0*rk_double(state) - 1.0;
393  x2 = 2.0*rk_double(state) - 1.0;
394  r2 = x1*x1 + x2*x2;
395  }
396  while (r2 >= 1.0 || r2 == 0.0);
397 
398  /* Box-Muller transform */
399  f = sqrt(-2.0*log(r2)/r2);
400  /* Keep for next call */
401  state->gauss = f*x1;
402  state->has_gauss = 1;
403  return f*x2;
404  }
405 }
